DataMapper Pattern

Conceitos Fundamentais
•

Objetivo: Separar totalmente lógica de domínio da persistência

•

Função: Mapeamento entre objeto de domínio e objeto de persistência

•

Característica: Domínio não tem setters, apenas getters

•

Integração: Pode ser usado junto ao DAO

Diferenças do DAO
DAO vs DataMapper
• DAO: Mistura mapeamento com SQL
• DataMapper: Separa mapeamento em método específico
• DAO: Usa setters na entidade
• DataMapper: Usa construtor/factory methods
Vantagens
•

Domínio puro sem dependência de persistência

•

Facilita testes unitários

•

Reutilização de mapeamento

User.java - Domínio Puro
public class User {
    private Long id;
    private String name;
    private String email;

    public User(Long id, String name, String email) {
        this.id = id;
        this.name = name;
        this.email = email;
    }

    public void changeEmail(String newEmail) {
        if (newEmail == null || !newEmail.contains("@"))
            throw new IllegalArgumentException("invalid email");
        this.email = newEmail;
    }

    // getters (no setters for invariantes de negócio)
    public Long getId() { return id; }
    public String getName() { return name; }
    public String getEmail() { return email; }
}
UserMapper.java - Interface
public interface UserMapper {
    User load(Long id) throws SQLException;
    void insert(User user) throws SQLException;
    void update(User user) throws SQLException;
}
UserMapperJdbc.java - Implementação
import javax.sql.DataSource;
import java.sql.*;

public class UserMapperJdbc implements UserMapper {
    private final DataSource ds;

    public UserMapperJdbc(DataSource ds) {
        this.ds = ds;
    }

    @Override
    public User load(Long id) throws SQLException {
        String sql = "SELECT id, name, email FROM users WHERE id = ?";
        try (Connection c = ds.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setLong(1, id);
            try (ResultSet rs = ps.executeQuery()) {
                if (!rs.next()) return null;
                return mapRowToUser(rs);
            }
        }
    }

    @Override
    public void insert(User user) throws SQLException {
        String sql = "INSERT INTO users (name,email) VALUES (?, ?)";
        try (Connection c = ds.getConnection();
             PreparedStatement ps = c.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {
            ps.setString(1, user.getName());
            ps.setString(2, user.getEmail());
            ps.executeUpdate();
        }
    }

    @Override
    public void update(User user) throws SQLException {
        String sql = "UPDATE users SET name = ?, email = ? WHERE id = ?";
        try (Connection c = ds.getConnection();
             PreparedStatement ps = c.prepareStatement(sql)) {
            ps.setString(1, user.getName());
            ps.setString(2, user.getEmail());
            ps.setLong(3, user.getId());
            ps.executeUpdate();
        }
    }

    private User mapRowToUser(ResultSet rs) throws SQLException {
        return new User(
            rs.getLong("id"),
            rs.getString("name"),
            rs.getString("email")
        );
    }
}
UserServiceWithMapper.java - Uso
public class UserServiceWithMapper {
    private final UserMapper mapper;

    public UserServiceWithMapper(UserMapper mapper) {
        this.mapper = mapper;
    }

    public void changeEmail(Long id, String newEmail) throws SQLException {
        User user = mapper.load(id);
        if (user == null) throw new RuntimeException("not found");
        user.changeEmail(newEmail); // lógica de negócio encapsulada na entidade
        mapper.update(user);
    }
}
